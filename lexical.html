int isKeyword(char word[]) {
    char keywords[][10] = {
        "int", "float", "if", "else", "while", "return", "for", "char", "double"
    };

    for (int i = 0; i < 9; i++) {
        if (strcmp(word, keywords[i]) == 0)
            return 1;
    }
    return 0;
}

int main() {
    char input[100], word[50];
    int i = 0, j;

    printf("Enter source code line:\n");
    fgets(input, sizeof(input), stdin);

    while (input[i] != '\0') {

        // If alphabet -> Identifier or Keyword
        if (isalpha(input[i])) {
            j = 0;
            while (isalnum(input[i])) {
                word[j++] = input[i++];
            }
            word[j] = '\0';

            if (isKeyword(word))
                printf("%s -> Keyword\n", word);
            else
                printf("%s -> Identifier\n", word);
        }

        // If digit -> Number
        else if (isdigit(input[i])) {
            j = 0;
            while (isdigit(input[i])) {
                word[j++] = input[i++];
            }
            word[j] = '\0';
            printf("%s -> Number\n", word);
        }

        // Operators
        else if (strchr("+-*/=<>", input[i])) {
            printf("%c -> Operator\n", input[i]);
            i++;
        }

        // Special Symbols
        else if (strchr(";,(){}[]", input[i])) {
            printf("%c -> Special Symbol\n", input[i]);
            i++;
        }

        else {
            i++;  // Skip spaces or unknown characters
        }
    }

    return 0;
}
